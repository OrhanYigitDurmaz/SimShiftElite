#include <Arduino.h>
#include <EEPROM.h>

struct MyData {
  int value1;
  int value2;
};

MyData data1 = {4094, 4095};
MyData data2 = {2304, 2305};

void setup() {
  EEPROM.begin(512);
  Serial.begin(115200);
  EEPROM.put(0, data1);
  EEPROM.commit();

}

void loop() {

  EEPROM.get(0, data1);
  Serial.println(data1.value1);

}
