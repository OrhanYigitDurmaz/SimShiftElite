#include <Arduino.h>
#include <EEPROM.h>

struct MyData {
  int value1;
  int value2;
};

MyData data[8] = {
  {4094, 4095},
  {2304, 2305},
  // Add six more instances with different values
  {1111, 2222},
  {3333, 4444},
  {5555, 6666},
  {7777, 8888},
  {9999, 1010},
  {1212, 1313}
};

void setup() {
  EEPROM.begin(512);
  Serial.begin(115200);
  
  // Write the array of data to EEPROM
  for (int i = 0; i < 8; i++) {
    EEPROM.put(i * sizeof(MyData), data[i]);
  }
  EEPROM.commit();
}

void loop() {
  // Read and print the data stored in EEPROM
  for (int i = 0; i < 8; i++) {
    MyData EEPROMData;
    EEPROM.get(i * sizeof(MyData), EEPROMData);
    Serial.print("Value1: ");
    Serial.print(EEPROMData.value1);
    Serial.print(", Value2: ");
    Serial.println(EEPROMData.value2);
    delay(1000); // Add a delay to avoid flooding the serial monitor with data
  }
}
