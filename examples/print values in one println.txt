#include <Arduino.h>
#include <EEPROM.h>

struct MyData {
  int value1;
  int value2;
};

MyData data1 = {4094, 4095};
MyData data2 = {2304, 2305};

void setup() {
  EEPROM.begin(512);
  Serial.begin(115200);
  EEPROM.put(0, data1);
  EEPROM.put(sizeof(MyData), data2);
  EEPROM.commit();

}

void loop() {

  MyData EEPROMData = EEPROM.get(0, data1);
  MyData EEPROMData2 = EEPROM.get(sizeof(MyData), data2);
  String printingData = String(EEPROMData.value1) + ", " + String(EEPROMData.value2);
  String printingData2 = String(EEPROMData2.value1) + ", " + String(EEPROMData2.value2);

  Serial.println(printingData);
  Serial.println(printingData2);
}
